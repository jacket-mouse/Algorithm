# 电话号码的字母组合

https://leetcode.cn/problems/letter-combinations-of-a-phone-number/description/



```c++
class Solution {
private:
    vector<string> code;
    string path;
    vector<string> ans;
    void init(){
        code.push_back("");
        code.push_back("");
        code.push_back("abc");
        code.push_back("def");
        code.push_back("ghi");
        code.push_back("jkl");
        code.push_back("mno");
        code.push_back("pqrs");
        code.push_back("tuv");
        code.push_back("wxyz");
        path = "";
    }
    void backtracing(string digits, int len, int start){
        if(path.size() == len){
            ans.push_back(path);
            return;
        }
        int index = (int)(digits.at(start)-'0');
        string s = code[index];
        for(int i = 0; i < s.size(); i++){
            path.push_back(s.at(i));
            backtracing(digits, len, start+1);
            path.pop_back();
        }
    }
public:
    vector<string> letterCombinations(string digits) {
        init();
        backtracing(digits, digits.size(), 0);
        return ans;
    }
};
```
