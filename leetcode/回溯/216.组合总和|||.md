# 组合总和|||

下面是一个错解，没有用到 k，把所有组合之和为 n 都进行了返回。但是题目要求只有 k 个数的组合是有效的。

```c++
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> path;
    int sum = 0;
    void backtracing(int k, int n, int start){
        if(sum == n){
            ans.push_back(path);
            return;
        }else if(sum > n) return;

        for(int i = start; i <= 9; i++){
            path.push_back(i);
            sum += i;
            backtracing(k, n, i+1);
            path.pop_back();
            sum -= i;
        }
    }
public:
    vector<vector<int>> combinationSum3(int k, int n) {
        backtracing(k, n, 1);
        return ans;
    }
};
```

```c++
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> path;
    int sum = 0;
    void backtracing(int k, int n, int start){
        if(sum == n && path.size() == k){
            ans.push_back(path);
            return;
        }else if(sum > n || path.size() > k) return;

        for(int i = start; i <= 9; i++){
            path.push_back(i);
            sum += i;
            backtracing(k, n, i+1);
            path.pop_back();
            sum -= i;
        }
    }
public:
    vector<vector<int>> combinationSum3(int k, int n) {
        backtracing(k, n, 1);
        return ans;
    }
};
```

C 语言版本

注意`pathTop`和一开始分配`path`空间的关系。

```c
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int** ans;
int ansTop;
int* path;
int pathTop;
int sum;
void backtracing(int k, int n, int start){
    if(sum == n && pathTop == k){
        int* tmp = malloc(k * sizeof(int));
        for(int i = 0; i < k; i++) tmp[i] = path[i];
        ans[ansTop++] = tmp;
        return;
    }else if(sum > n || pathTop >= k) return; // pathTop > k 下面分配空间就要 k+1 个

    for(int i = start; i <= 9; i++){
        path[pathTop++] = i;
        sum += i;
        backtracing(k, n, i+1);
        pathTop--;
        sum -= i;
    }
}
int** combinationSum3(int k, int n, int* returnSize, int** returnColumnSizes) {
    path = (int*)malloc(k * sizeof(int));
    ans = (int**)malloc(1000 * sizeof(int*));
    ansTop = pathTop = sum = 0;
    backtracing(k, n, 1);
    *returnSize = ansTop; // 标志着 ans 数组里有多少个一维数组(path)
    *returnColumnSizes = (int*)malloc(sizeof(int) * (*returnSize)); // 标志着每个 path 里有多少元素
    for(int i = 0; i < ansTop; i++){
        (*returnColumnSizes)[i] = k;
    }
    return ans;
}
```
