# 两数之和

https://leetcode.cn/problems/two-sum/

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> ans;
        unordered_map<int, int> hashtable;
        for (int i = 0; i < nums.size(); ++i) {
            int complement = target - nums[i];
            auto it = hashtable.find(complement);
            if (it != hashtable.end()) {
                return {it->second, i};
            }
            hashtable[nums[i]] = i;
        }
        return {};
    }
};
```
